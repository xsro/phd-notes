\section{双积分系统}

\subsection{问题描述}

考虑 $N\ge 3$ 个机器人，编号集合为 $\mathcal{N}=\{1,2,\ldots,N\}$，每个机器人采用双积分模型
$\dot p_i=v_i,\quad \dot v_i=u_i+d_i,\quad i\in\mathcal{N},$
其中 $p_i,v_i,u_i,d_i\in\mathbb{R}^2$ 分别为位置、速度、控制输入和输入通道扰动。目标编号为 $0$，位置为 $p_0(t)\in\mathbb{R}^2$。控制目标为：在保持通信连通和避免机器人间碰撞的前提下，使目标进入机器人位置凸包。固定时间增强器进一步提升质心通道的收敛速度；若进入质心估计器收敛条件的时间依赖初始状态，则整体结论为有限时间合围。

\subsection{已有算法}

整体控制方案由四部分组成：首先设计分布式有限时间观测器估计目标状态，然后构造积分型人工势场处理约束，接着定义复合误差并将其纳入 RISE 框架，最终给出完整的合围控制律。

\subsubsection{分布式有限时间观测器}

令 $\hat p_{i0}$ 和 $\hat v_{i0}$ 分别表示相对位置 $p_{i0}=p_i-p_0$ 和相对速度 $\dot p_{i0}=\dot p_i-\dot p_0$ 的估计值，定义估计误差 $e_i=\hat p_{i0}-p_{i0}$，并构造一致性变量
$\xi_i=\sum_{j\in\mathcal{N}_i(0)}a_{ij}(e_i-e_j)+b_i e_i.$
实际实现时 $\xi_i$ 可用相对测量表示为
$\xi_i=\sum_{j\in\mathcal{N}_i(0)}a_{ij}(\hat p_{i0}-\hat p_{j0}-p_{ij})+b_i(\hat p_{i0}-p_{i0}),$
其中最后一项仅对 $b_i>0$ 的机器人存在。

基于 $\xi_i$，设计如下有限时间观测器：
$$
\begin{cases}
\dot{\hat p}_{i0}=\hat v_{i0}-k_a\lfloor\xi_i\rceil^{1/2},\\
\dot{\hat v}_{i0}=u_i-k_b\operatorname{sgn}(\xi_i),
\end{cases}
\quad i\in\mathcal{N},
$$
其中 $\operatorname{sgn}(x)=x/\|x\|$（$x=0$ 时取 $\mathbf{0}$），$\lfloor x\rceil^q=\|x\|^q\operatorname{sgn}(x)$。参数选取满足
$k_b\ge \bar n/\rho,\quad k_a\ge \bigl((2\gamma_0+2\gamma_1+1)/\lambda_1\bigr)^{1/2}k_b^{1/2},$
其中 $0<\rho<1$，$\gamma_1=1+\rho$，$\gamma_0=(1+3\gamma_1)/(1-\rho)$，$\lambda_1$ 为 $\mathcal{M}(0)=\mathcal{L}(0)+\mathcal{B}$ 的最小特征值，扰动上界为 $\bar n=\max_{i\in\mathcal{N}}\bar n_i$，$\bar n_i=\varrho_2+\bar d_{0i}$。

\subsubsection{积分型人工势场}

为同时实现避碰与初始通信边保持，引入积分型人工势场（iAPF）$\phi_i=\phi_i^r+\phi_i^p$。其中避碰项为
$\phi_i^r=\sum_{j\in\mathcal{N}_i(t)}\alpha_r(\|p_{ij}\|)\frac{p_{ij}}{\|p_{ij}\|},$
函数 $\alpha_r(s)$ 在 $s\to d^+$ 时趋于 $+\infty$，在 $s\ge \mu$ 时为 $0$。连通性保持项为
$\phi_i^p=\sum_{j\in\mathcal{N}_i(0)}\alpha_p(\|p_{ij}\|)\frac{p_{ij}}{\|p_{ij}\|},$
函数 $\alpha_p(s)$ 在 $s\to \mu^-$ 时趋于 $-\infty$，在 $s\to d^+$ 时趋于 $0$。

一组可选的具体函数为
$$
\alpha_r(s)=
\begin{cases}
a_r(s)-a_r(l_d), & s\in(d,l_d),\\
0, & s\in[l_d,\infty),
\end{cases}
\quad
a_r(s)=\frac{c_r}{s-d},
$$
$$
\alpha_p(s)=
\begin{cases}
0, & s\in(d,l_\mu),\\
a_p(s)-a_p(l_\mu), & s\in[l_\mu,\mu),
\end{cases}
\quad
a_p(s)=\frac{c_p}{s-\mu},
$$
其中 $l_d,l_\mu\in(d,\mu)$，$c_r,c_p>0$。

\subsubsection{SMC 合围控制律}

引入 iAPF 辅助变量 $\Phi_{1i}$，满足 $\dot\Phi_{1i}=k_1p_{i0}-\phi_i$，$\Phi_{1i}(0)=\mathbf{0}$。在此基础上定义复合误差
$s_{i0}=\dot p_{i0}+k_2p_{i0}+\Phi_{1i}.$
该构造将相对速度误差、相对位置误差和势场补偿统一到一个变量中。由于 $p_{i0}$ 与 $\dot p_{i0}$ 不完全可测，用观测量替代得到
$\hat s_{i0}=\hat v_{i0}+k_2\hat p_{i0}+\hat\Phi_{1i},$
其中 $\hat\Phi_{1i}$ 按相同方式在线积分：$\dot{\hat\Phi}_{1i}=k_1\hat p_{i0}-\phi_i$，$\hat\Phi_{1i}(0)=\mathbf{0}$。

基于上述复合误差，设计控制输入 $u_i=-k_s \mathrm{sgn}(\hat s_{i0})+\hat\Delta_i$，其中 $k_s>0$，$\alpha>0$，$\beta_i>0$。补偿项 $\hat\Delta_i$ 用于抵消已知的动力学分量：
$\hat\Delta_i=-k_2\hat v_{i0}-k_1\hat p_{i0}+\phi_i,$
对应的不含观测误差的真实量为 $\Delta_i=-k_2\dot p_{i0}-k_1p_{i0}+\phi_i$。
增益选取满足
$k_s \ge \sup_{t\geq 0}\|\ddot{p}_0\| + \sup_{t\geq 0,i\in\mathbb{N}}\|d_i\|$.

\begin{lemma}[原 SMC--iAPF 算法的渐近性质]
在上述观测器、积分型人工势场和 SMC 控制律的增益条件下，原算法闭环满足 $\lim_{t\to\infty}\dot p_{i0}(t)=\mathbf{0}$、$\sup_{t\geq0}\|\ddot p_{i0}(t)\|<\infty$，并且 $\lim_{t\to\infty}(k_1p_{i0}(t)-\phi_i(t))=\mathbf{0}$，$i\in\mathcal{N}$。
\end{lemma}

\begin{proof}
该结论是原 SMC--iAPF 算法的闭环收敛性质。由观测器有限时间收敛和滑模变量收敛可得 $\dot p_{i0}$ 收敛；由闭环右端各项有界可得 $\ddot p_{i0}$ 有界。另一方面，辅助变量满足 $\dot\Phi_{1i}=k_1p_{i0}-\phi_i$，结合闭环收敛性可得 $\dot\Phi_{1i}\to\mathbf{0}$，从而 $k_1p_{i0}-\phi_i\to\mathbf{0}$。
\end{proof}


\subsection{固定时间质心增强器}

上述 SMC 控制律主要保证安全约束和渐近合围。固定时间增强器不直接要求所有 $p_{i0}\to\mathbf{0}$，因为这会与避碰约束冲突；增强器只作用在集群的平移自由度上，使相对目标的集群中心 $c$ 及其速度 $\nu=\dot c$ 在增强阶段固定时间到达零点，其中 $c=N^{-1}\sum_i p_{i0}$、$\nu=N^{-1}\sum_i\dot p_{i0}$。机器人之间的相对构型、避碰和连通性仍由原 iAPF--SMC 层负责。

为避免增强器在估计误差尚未消失时破坏相对动力学，采用延迟开启机制。记 $T_s(x(0))$ 为原算法进入质心估计器收敛条件的有限时间，该时间一般依赖初始状态。动态平均一致器在 $T_s$ 后初始化，中心增强控制在估计完成后开启。开启后每个机器人使用同一个公共平移加速度，因此增强项从所有相对动力学中精确抵消。

\subsubsection{增强器预备引理}

\begin{lemma}[观测器有限时间收敛]
在已有算法的观测器增益条件下，存在已知常数 $T_o>0$，使得对所有 $t\geq T_o$，有 $\hat p_{i0}(t)=p_{i0}(t)$、$\hat v_{i0}(t)=\dot p_{i0}(t)$，$i\in\mathcal{N}$。
\end{lemma}

本文直接使用上述观测器结论，不再重复其证明。

\begin{lemma}[固定时间 Lyapunov 判据]
若正定函数 $V$ 满足 $\dot V\leq-aV^m-bV^n$，其中 $a,b>0$ 且 $0<m<1<n$，则 $V$ 在固定时间内收敛到零，且收敛时间满足 $T\leq 1/[a(1-m)]+1/[b(n-1)]$。
\end{lemma}

\begin{proof}
当 $V\geq1$ 时，$\dot V\leq-bV^n$，因此从任意初值进入集合 $V\leq1$ 的时间不超过 $1/[b(n-1)]$。当 $0<V\leq1$ 时，$\dot V\leq-aV^m$，从 $V=1$ 到 $V=0$ 的时间不超过 $1/[a(1-m)]$。两段时间上界与初值无关，故为固定时间收敛。
\end{proof}

\begin{lemma}[动态平均一致]
考虑无向连通图。对每个可微的信号 $r_i(t)\in\mathbb{R}^2$，从任意时刻 $t_0$ 初始化状态 $z_i(t_0)=r_i(t_0)$，并令
$$
\dot z_i=\dot r_i
-\kappa_{1}\sum_{j\in\mathcal{N}_i(0)}a_{ij}\lfloor z_i-z_j\rceil^{\alpha}
-\kappa_{2}\sum_{j\in\mathcal{N}_i(0)}a_{ij}\lfloor z_i-z_j\rceil^{\beta}
-\kappa_{3}\sum_{j\in\mathcal{N}_i(0)}a_{ij}\operatorname{sgn}(z_i-z_j),
$$
其中 $0<\alpha<1$，$\beta>1$。若存在已知常数 $\bar r$ 使得 $\|\dot r_i(t)-\dot r_j(t)\|\leq\bar r$，$\forall i,j$，$\forall t\geq t_0$，则当 $\kappa_3$ 足够大时，存在与初值无关的 $T_a>0$，使得 $z_i(t)=N^{-1}\sum_{j=1}^{N}r_j(t)$，$\forall i\in\mathcal{N}$，$\forall t\geq t_0+T_a$。
\end{lemma}

\begin{proof}
由图无向性可知所有边上非线性一致项求和为零，因此若 $z_i(t_0)=r_i(t_0)$，则 $N^{-1}\sum_{i=1}^{N}z_i(t)=N^{-1}\sum_{i=1}^{N}r_i(t)$ 对所有 $t\geq t_0$ 成立。令 $\bar z=N^{-1}\sum_i z_i$、$\varepsilon_i=z_i-\bar z$，并对不一致能量 $V_a=\frac{1}{2}\sum_i\|\varepsilon_i\|^2$ 求导。连通图的边差范数与节点不一致范数等价，因此两类齐次一致项分别给出 $V_a^{(1+\alpha)/2}$ 和 $V_a^{(1+\beta)/2}$ 型耗散；输入差 $\dot r_i-\dot r_j$ 产生的项被符号一致项吸收。选取 $\kappa_3$ 大于由图权重和 $\bar r$ 决定的阈值后，可得
$$
\dot V_a\leq
-a_a V_a^{(1+\alpha)/2}
-b_a V_a^{(1+\beta)/2},
$$
其中 $a_a,b_a>0$。由固定时间 Lyapunov 判据，$V_a$ 在与初值无关的时间 $T_a$ 内到达零。再结合平均值守恒，得到结论。
\end{proof}

\begin{lemma}[动态平均输入有界性]
在原 SMC--iAPF 算法的渐近性质引理成立时，对给定初始状态，存在有限时间 $T_s(x(0))\geq T_o$ 和常数 $\bar r_p,\bar r_v>0$，使得对所有 $i,j$ 和 $t\geq T_s(x(0))$，有 $\|\dot r_i^p(t)-\dot r_j^p(t)\|\leq\bar r_p$ 和 $\|\dot r_i^v(t)-\dot r_j^v(t)\|\leq\bar r_v$。
\end{lemma}

\begin{proof}
由观测器有限时间收敛引理，对所有 $t\geq T_o$，有 $r_i^p=\hat p_{i0}=p_{i0}$ 和 $r_i^v=\hat v_{i0}=\dot p_{i0}$。因此位置平均一致器的输入导数满足 $\dot r_i^p-\dot r_j^p=\dot p_{i0}-\dot p_{j0}$。
由原 SMC--iAPF 算法的渐近性质引理，$\dot p_{i0}$ 收敛，故对给定初始状态，存在有限时间 $T_{sp}(x(0))$ 使任意差分 $\dot p_{i0}-\dot p_{j0}$ 在 $t\geq T_{sp}$ 后满足所需上界，记该上界为 $\bar r_p$。

速度平均一致器的输入导数满足 $\dot r_i^v-\dot r_j^v=\ddot p_{i0}-\ddot p_{j0}$。
由原 SMC--iAPF 算法的渐近性质引理，$\ddot p_{i0}$ 有界，故任意差分 $\ddot p_{i0}-\ddot p_{j0}$ 有界，存在 $\bar r_v>0$ 使第二式成立。取 $T_s(x(0))=\max\{T_o,T_{sp}(x(0))\}$ 即得结论。由于 $T_{sp}$ 来自 $\dot p_{i0}$ 的渐近收敛，它一般依赖初始状态。
\end{proof}

\begin{lemma}[鲁棒固定时间中心控制]
考虑受扰双积分中心系统 $\dot c=\nu$、$\dot\nu=a+w(t)$，其中 $\|w(t)\|\leq\bar w$。存在固定时间反馈 $a=a_{\mathrm{FxT}}(c,\nu)$，使得 $c$ 与 $\nu$ 在固定时间内收敛到零。一个可选的理想滑模实现如下。定义 $\sigma=\nu+\lambda_1\lfloor c\rceil^{\gamma}+\lambda_2\lfloor c\rceil^{\delta}$，其中 $0<\gamma<1<\delta$，$\lambda_1,\lambda_2>0$。令
$$
\begin{aligned}
a_{\mathrm{FxT}}
={}&-\lambda_1\frac{d}{dt}\lfloor c\rceil^\gamma
-\lambda_2\frac{d}{dt}\lfloor c\rceil^\delta\\
&-\ell_1\lfloor\sigma\rceil^{\rho}
-\ell_2\lfloor\sigma\rceil^{\eta}
-k_w\operatorname{sgn}(\sigma),
\end{aligned}
$$
其中 $0<\rho<1<\eta$，$\ell_1,\ell_2>0$，$k_w>\bar w$。则 $\sigma$ 固定时间收敛到零；随后 $c$ 与 $\nu$ 固定时间收敛到零。
\end{lemma}

\begin{proof}
代入控制律得到 $\dot\sigma=-\ell_1\lfloor\sigma\rceil^{\rho}-\ell_2\lfloor\sigma\rceil^{\eta}-k_w\operatorname{sgn}(\sigma)+w$。取 $V_\sigma=\frac{1}{2}\|\sigma\|^2$，则
$$
\dot V_\sigma
\leq
-\ell_1\|\sigma\|^{1+\rho}
-\ell_2\|\sigma\|^{1+\eta}
-(k_w-\bar w)\|\sigma\|.
$$
忽略最后一个非正项，并把 $\|\sigma\|=(2V_\sigma)^{1/2}$ 代入，可得
$$
\dot V_\sigma
\leq
-a_\sigma V_\sigma^{(1+\rho)/2}
-b_\sigma V_\sigma^{(1+\eta)/2},
$$
其中 $a_\sigma,b_\sigma>0$。由固定时间 Lyapunov 判据，$\sigma$ 在固定时间内到达零。

由于 $k_w>\bar w$，在 Filippov 意义下符号项能够提供保持滑模面所需的等效控制，因此到达 $\sigma=0$ 后该滑模面保持不变。此时由 $\sigma=0$ 得 $\dot c=-\lambda_1\lfloor c\rceil^\gamma-\lambda_2\lfloor c\rceil^\delta$。
取 $V_c=\frac{1}{2}\|c\|^2$，同理有
$$
\dot V_c
\leq
-a_c V_c^{(1+\gamma)/2}
-b_c V_c^{(1+\delta)/2},
$$
其中 $a_c,b_c>0$。再次使用固定时间 Lyapunov 判据，$c$ 固定时间收敛到零，且由滑模面 $\sigma=0$ 得 $\nu$ 同时收敛到零。
\end{proof}

\begin{remark}
上述中心控制律是便于证明的理想形式。由于 $0<\gamma<1$ 时 $\frac{d}{dt}\lfloor c\rceil^\gamma$ 在 $c=\mathbf{0}$ 附近可能带来奇异性，实际实现可采用非奇异终端滑模等价形式或在原点邻域使用光滑饱和近似；证明中使用的是标准 Filippov 意义下的滑模模型。
\end{remark}

\subsubsection{质心增强器设计}

理论分析采用方案 A：不要求机器人在线协商 $T_e$，而是把 $T_e$ 作为质心估计条件满足并完成估计后的分析时刻。该方案用于给出清晰的有限时间合围证明，但 $T_s(x(0))$ 和 $T_e$ 一般不是严格分布式可在线获得的量。

在 $t=T_s(x(0))$ 时初始化两个动态平均一致器：$z_i^p(T_s)=\hat p_{i0}(T_s)$、$z_i^v(T_s)=\hat v_{i0}(T_s)$。分别以 $r_i^p=\hat p_{i0}$ 和 $r_i^v=\hat v_{i0}$ 作为输入运行动态平均一致器，并定义 $\hat c_i=z_i^p$、$\hat\nu_i=z_i^v$。令 $T_e=T_s(x(0))+T_a$。由观测器有限时间收敛、动态平均输入有界性引理和动态平均一致引理可知，对所有 $t\geq T_e$，有 $\hat c_i=c$、$\hat\nu_i=\nu$，$\forall i\in\mathcal{N}$。
这里动态平均一致引理所需的输入变化率差有界条件由“动态平均输入有界性”引理给出；实际选取 $\kappa_3$ 时取其大于由 $\max\{\bar r_p,\bar r_v\}$ 和图权重决定的阈值即可。

中心增强项定义为
$$
a_i^c(t)=
\begin{cases}
\mathbf{0}, & t<T_e,\\
a_{\mathrm{FxT}}(\hat c_i,\hat\nu_i), & t\geq T_e.
\end{cases}
$$
完整控制输入为 $u_i=u_i^{\mathrm{SMC}}+a_i^c$。

\subsubsection{分布式开启机制}

实际分布式实现可采用方案 B：局部残差检测加 ready 信息扩散。机器人 $i$ 构造观测器残差 $\epsilon_i^o=\|\xi_i\|$，以及质心估计一致性残差
$$
\epsilon_i^c=\sum_{j\in\mathcal{N}_i(0)}a_{ij}\|z_i^p-z_j^p\|,\qquad
\epsilon_i^\nu=\sum_{j\in\mathcal{N}_i(0)}a_{ij}\|z_i^v-z_j^v\|.
$$
给定阈值 $\varepsilon_o,\varepsilon_c,\varepsilon_\nu>0$ 和驻留时间 $\tau_d>0$，若 $\epsilon_i^o\leq\varepsilon_o$、$\epsilon_i^c\leq\varepsilon_c$、$\epsilon_i^\nu\leq\varepsilon_\nu$ 在区间 $[t-\tau_d,t]$ 内持续成立，则节点 $i$ 标记自身为 ready。

为避免部分节点提前开启导致 $a_i^c-a_j^c\neq\mathbf{0}$，可让 ready 信息在图上传播。设 $\mathcal{S}_i(t)$ 为节点 $i$ 已知的 ready 节点集合，初始化时若节点 $i$ ready 则令 $i\in\mathcal{S}_i$；通信时按规则 $\mathcal{S}_i\leftarrow \mathcal{S}_i\cup\bigcup_{j\in\mathcal{N}_i(0)}\mathcal{S}_j$ 更新。当 $|\mathcal{S}_i|=N$ 时，节点 $i$ 知道全网均已 ready，再开启质心增强器。该机制需要节点 ID、网络规模 $N$ 和可靠通信；若还要求所有节点同步开启，则需全局时钟、通信轮次同步或图直径上界。若无法保证同步开启，则开启过渡阶段会产生短暂非公共输入，需要把 $a_i^c-a_j^c$ 作为有界扰动另行分析。

\subsubsection{有限时间合围结论}

\begin{assumption}
设 $\mathcal{E}(0)$ 为初始通信边集合，并定义安全连通集合
$$
\Omega=\{p:\ \|p_{ij}\|>d,\ \forall i\neq j;\ \|p_{ij}\|<\mu,\ \forall (i,j)\in\mathcal{E}(0)\}.
$$
原 SMC--iAPF 控制层在无中心增强项时使 $\Omega$ 正不变。
\end{assumption}

\begin{assumption}
对 $t\geq T_e$，中心通道中的等效扰动
$$
w(t)=
\frac{1}{N}\sum_{i=1}^{N}u_i^{\mathrm{SMC}}(t)
+\frac{1}{N}\sum_{i=1}^{N}d_i(t)
-\ddot p_0(t)
$$
存在已知上界 $\bar w$，即 $\|w(t)\|\leq\bar w$。
\end{assumption}

\begin{theorem}
对方案 A，在上述假设下，若 $\kappa_3$ 足够大且中心控制器中 $k_w>\bar w$，则增强后的控制律保持原有避碰和连通约束，并使目标在有限时间内进入机器人位置凸包。更具体地，存在时间 $T^\star(x(0))\leq T_s(x(0))+T_a+T_c$，使得对所有 $t\geq T^\star(x(0))$，有 $c(t)=\mathbf{0}$、$\nu(t)=\mathbf{0}$ 和 $\mathrm{dist}(p_0(t),\mathrm{co}(p(t)))=0$。由于 $T_s(x(0))$ 一般依赖初始状态，该整体结论是有限时间而非固定时间；从 $T_e$ 开始的质心增强阶段仍具有固定时间上界 $T_c$。
\end{theorem}

\begin{proof}
第一步，证明增强项不破坏相对安全层。当 $t<T_e$ 时，$a_i^c=\mathbf{0}$，闭环就是原 SMC--iAPF 系统，因此 $\Omega$ 的正不变性由原控制层保证，特别地 $p(T_e)\in\Omega$。当 $t\geq T_e$ 时，由观测器引理和动态平均一致引理，
$$
\hat c_i=c,\qquad \hat\nu_i=\nu,\qquad \forall i\in\mathcal{N}.
$$
于是所有机器人使用同一个增强输入 $a_i^c=a_{\mathrm{FxT}}(c,\nu)$。对任意 $i,j$，有 $a_i^c-a_j^c=\mathbf{0}$。
因此在相对动力学
$$
\ddot p_{ij}
=u_i-u_j+d_i-d_j
$$
中，公共增强项完全抵消。于是从 $T_e$ 开始，相对位置、避碰势函数和初始边保持势函数看到的控制差分与原 SMC--iAPF 层一致。由 $\Omega$ 对原相对闭环的正不变性，增强后的闭环仍保持避碰和连通约束。

第二步，证明质心增强阶段固定时间收敛。由 $c=N^{-1}\sum_i p_{i0}$ 得
$$
\ddot c
=\frac{1}{N}\sum_{i=1}^{N}u_i
+\frac{1}{N}\sum_{i=1}^{N}d_i-\ddot p_0 .
$$
当 $t\geq T_e$ 时，$u_i=u_i^{\mathrm{SMC}}+a_{\mathrm{FxT}}(c,\nu)$，因此
$$
\dot c=\nu,\qquad
\dot\nu=a_{\mathrm{FxT}}(c,\nu)+w(t),
$$
其中 $w(t)$ 正是第二个假设中的有界扰动。由鲁棒固定时间中心控制引理，$c$ 和 $\nu$ 在一个与初始值无关的时间 $T_c$ 内收敛到零。

第三步，证明合围性质。对所有 $t\geq T_e+T_c$，有 $c(t)=N^{-1}\sum_{i=1}^{N}(p_i(t)-p_0(t))=\mathbf{0}$，于是 $p_0(t)=N^{-1}\sum_{i=1}^{N}p_i(t)$。由于等权平均点属于 $\mathrm{co}(p(t))$，得到 $\mathrm{dist}(p_0(t),\mathrm{co}(p(t)))=0$。
综上，取 $T^\star(x(0))=T_s(x(0))+T_a+T_c$ 即得有限时间结论。若能进一步给出与初始状态无关的 $T_s$ 上界，则整体结论才可升级为固定时间。
\end{proof}
